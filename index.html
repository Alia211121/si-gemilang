function bukaTab(evt, tabName) { // 1. Sembunyikan semua isi halaman terlebih dahulu document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden')); // 2. Kembalikan semua warna tombol menu menjadi abu-abu pasif document.querySelectorAll('.tab-link').forEach(el => { el.className = "tab-link flex-1 bg-slate-300/60 text-slate-700 font-bold py-3 px-4 rounded-xl hover:bg-slate-300 transition-all cursor-pointer"; }); // 3. Tampilkan isi halaman yang dipilih (Keuangan / SPP / RAB) document.getElementById(tabName).classList.remove('hidden'); // 4. PERBAIKAN: Logika pewarnaan tombol hijau aktif agar tidak macet lagi if(evt) { evt.currentTarget.className = "tab-link flex-1 bg-emerald-800 text-white font-bold py-3 px-4 rounded-xl shadow-xs transition-all cursor-pointer"; } else { document.getElementById('btn-' + tabName).className = "tab-link flex-1 bg-emerald-800 text-white font-bold py-3 px-4 rounded-xl shadow-xs transition-all cursor-pointer"; } updateSistemTampilan(); }
